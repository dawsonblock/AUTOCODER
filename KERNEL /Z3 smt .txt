import ast
import z3

class Z3EquivalenceChecker:
    """Pre-filters mathematically redundant AST patches."""
    def __init__(self):
        self.symbols = {}

    def _get_z3_var(self, name: str):
        if name not in self.symbols:
            self.symbols[name] = z3.Int(name)
        return self.symbols[name]

    def _ast_to_z3(self, node):
        if isinstance(node, ast.Name): return self._get_z3_var(node.id)
        elif isinstance(node, ast.Constant):
            if isinstance(node.value, int): return z3.IntVal(node.value)
            raise ValueError()
        elif isinstance(node, ast.BinOp):
            left, right = self._ast_to_z3(node.left), self._ast_to_z3(node.right)
            if isinstance(node.op, ast.Add): return left + right
            if isinstance(node.op, ast.Sub): return left - right
        elif isinstance(node, ast.Compare):
            left, right = self._ast_to_z3(node.left), self._ast_to_z3(node.comparators[0])
            op = node.ops[0]
            if isinstance(op, ast.Lt): return left < right
            if isinstance(op, ast.LtE): return left <= right
            if isinstance(op, ast.Gt): return left > right
            if isinstance(op, ast.GtE): return left >= right
            if isinstance(op, ast.Eq): return left == right
            if isinstance(op, ast.NotEq): return left != right
        raise ValueError()

    def is_semantically_equivalent(self, orig_code: str, mut_code: str) -> bool:
        try:
            orig_ast = ast.parse(orig_code, mode='eval').body
            mut_ast = ast.parse(mut_code, mode='eval').body
            self.symbols = {}
            z3_orig, z3_mut = self._ast_to_z3(orig_ast), self._ast_to_z3(mut_ast)
            
            solver = z3.Solver()
            solver.add(z3_orig != z3_mut) # Try to find a case where they diverge
            return solver.check() == z3.unsat # UNSAT means no counter-example exists (they are equivalent)
        except Exception:
            return False # Fail open, let VM test it
