#!/usr/bin/env python3
import subprocess
import sys
import time
import json
from typing import List

def run_cmd(cmd: List[str], timeout: int = 10) -> bool:
    try:
        res = subprocess.run(cmd, capture_output=True, text=True, timeout=timeout)
        return res.returncode == 0
    except subprocess.TimeoutExpired:
        return False

def execute_funnel(target_file: str, target_test: str) -> dict:
    start_time = time.time()
    
    # Stage 1: Static Analysis
    if not run_cmd(["ruff", "check", target_file, "--select", "E,F"]):
        return {"success": False, "tests_failed": 999, "error": "LintError", "runtime": time.time() - start_time}
    if not run_cmd(["mypy", target_file, "--ignore-missing-imports"]):
        return {"success": False, "tests_failed": 999, "error": "TypeError", "runtime": time.time() - start_time}
        
    # Stage 2: Targeted Test
    if not run_cmd(["pytest", target_test, "-q", "-x", "--disable-warnings"]):
        return {"success": False, "tests_failed": 1, "error": "TargetTestFailed", "runtime": time.time() - start_time}
        
    # Stage 3: Full Suite
    full_suite_passed = run_cmd(["pytest", target_test, "-q", "--disable-warnings"], timeout=30)
    
    if full_suite_passed:
        return {"success": True, "tests_failed": 0, "error": None, "runtime": time.time() - start_time}
    else:
        return {"success": False, "tests_failed": 1, "error": "RegressionError", "runtime": time.time() - start_time}

if __name__ == "__main__":
    if len(sys.argv) < 3:
        print(json.dumps({"success": False, "tests_failed": 999, "error": "MissingArgs", "runtime": 0}))
        sys.exit(1)
        
    print(json.dumps(execute_funnel(sys.argv[1], sys.argv[2])))
